    <div id="game-container">

        <h1>Gioco di Distrazione</h1>

        <p id="instructions">Tappa il pulsante esattamente quando il cronometro segna un cambio di secondo (es. 1.000, 2.000, 3.000...)</p>

        <div id="timer">0.000</div>

        <div id="score-display">Punteggio: 0</div>

        <button id="tap-button">Tappa!</button>

        <div id="results"></div>

    </div>


    <script>

        const timerDisplay = document.getElementById('timer');

        const scoreDisplay = document.getElementById('score-display');

        const tapButton = document.getElementById('tap-button');

        const resultsDisplay = document.getElementById('results');


        let startTime = 0;

        let timerInterval = null;

        let score = 0;


        function startTimer() {

            startTime = Date.now();

            timerInterval = setInterval(() => {

                const elapsedTime = (Date.now() - startTime) / 1000;

                timerDisplay.textContent = elapsedTime.toFixed(3);

            }, 10);

        }


        function stopTimer() {

            clearInterval(timerInterval);

            timerInterval = null;

        }


        function resetGame() {

            stopTimer();

            timerDisplay.textContent = '0.000';

            resultsDisplay.textContent = '';

            startTimer();

        }


        function calculateScore(time) {

            const decimalPart = time - Math.floor(time);

            const deviation = Math.abs(decimalPart - 0); // L'obiettivo è essere il più vicino a .000

            

            if (deviation < 0.05) { // Eccellente (entro 50ms)

                return 10;

            } else if (deviation < 0.1) { // Buono (entro 100ms)

                return 5;

            } else if (deviation < 0.2) { // Discreto (entro 200ms)

                return 2;

            }

            return 0; // Fuori tempo

        }


        tapButton.addEventListener('click', () => {

            if (!timerInterval) {

                startTimer();

                return;

            }

            

            stopTimer();

            const tapTime = parseFloat(timerDisplay.textContent);

            const points = calculateScore(tapTime);

            

            score += points;

            scoreDisplay.textContent = `Punteggio: ${score}`;


            if (points > 0) {

                resultsDisplay.textContent = `Hai beccato il tempo! Punti: +${points}`;

                resultsDisplay.style.color = '#28a745';

            } else {

                resultsDisplay.textContent = `Ritardo...`;

                resultsDisplay.style.color = '#dc3545';

            }

            

            setTimeout(resetGame, 1500); // Resetta dopo 1.5 secondi per dare il tempo di vedere il risultato

        });


        // Avvia il gioco al primo click sul pulsante

        tapButton.textContent = "Inizia!";

    </script>
